#include <stdio.h>

int main()
{
        for (int i = 1; i <= 9; i++)
    {
        for (int j = 2; j <= 9; j++)
        {
            printf("%3d x %d = %2d", j, i, (i*j));
        }
        printf("\n");
    }
    return 0;
}

{

    int random = (rand() % 3)+3; //
    
    printf("%d", random);
    for(int i=0;i<random;i++){
    for(int j=0;j<=i;j++){
	printf("*");
    }
    printf("\n");
}

}
